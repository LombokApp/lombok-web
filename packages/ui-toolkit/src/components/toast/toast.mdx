import { 
  Button, 
  useToast, 
  Toast, 
  ToastAction,
  Toaster
} from '@stellariscloud/ui-toolkit'

const { toast } = useToast()

<div className="flex h-[200px] w-[400px] flex-col items-center justify-center">
  <Toaster />
  <Button
    variant="outline"
    onClick={() => {
      toast({
        description: 'Your message has been sent.',
      })
    }}
  >
    Launch toast
  </Button>
</div>
`} />
